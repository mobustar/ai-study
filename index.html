# ── Lasso 回帰 (L1 正則化) — 座標降下法 ─────────
def lasso_fit(Xb, y, alpha, max_iter=200):
    w = np.zeros(Xb.shape[1])
    for _ in range(max_iter):
        for j in range(len(w)):
            residual = y - Xb @ w + Xb[:, j] * w[j]  # j以外の寄与を除く
            rho = Xb[:, j] @ residual                  # ρ = X_j · r
            z_j = np.sum(Xb[:, j] ** 2)               # ||X_j||²
            # Soft-thresholding (L1 の近接オペレータ)
            w[j] = np.sign(rho) * max(abs(rho) - alpha, 0) / z_j
    return w

# ── Batch Normalization ───────────────────────────
class BatchNorm:
    def forward(self, x):
        if self.training:
            mu  = x.mean(axis=0)          # ミニバッチ平均
            var = x.var(axis=0)           # ミニバッチ分散
            self.x_hat = (x - mu) / np.sqrt(var + self.eps)
            # 移動平均を更新 → 推論時に使用
            self.running_mean = 0.9 * self.running_mean + 0.1 * mu
            self.running_var  = 0.9 * self.running_var  + 0.1 * var
        else:
            self.x_hat = (x - self.running_mean) / np.sqrt(self.running_var + self.eps)
        return self.gamma * self.x_hat + self.beta  # γ・β でスケール・シフト

# ── Inverted Dropout ──────────────────────────────
class Dropout:
    def forward(self, x):
        if self.training:
            # 1/(1-p) をかけて推論時に何もしなくて済むようにする
            self.mask = np.random.binomial(1, 1-self.p, x.shape) / (1-self.p)
            return x * self.mask
        return x     # 推論時はそのまま通す

# ── モデル構成 (計算グラフ) ───────────────────────

# ── im2col: 畳み込み → GEMM 変換 ─────────────────
def im2col(x, kh, kw, stride=1, pad=0):
    """入力 (N,C,H,W) → (N·oh·ow, C·kh·kw) の行列に展開"""
    N, C, H, W = x.shape
    x_pad = np.pad(x, ((0,0),(0,0),(pad,pad),(pad,pad)))
    oh = (H + 2*pad - kh) // stride + 1  # 出力サイズ計算
    ow = (W + 2*pad - kw) // stride + 1
    col = np.zeros((N, C, kh, kw, oh, ow))
    for i in range(kh):
        for j in range(kw):
            col[:,:,i,j,:,:] = x_pad[:,:, i::stride, j::stride][:,:,:oh,:ow]
    return col.transpose(0,4,5,1,2,3).reshape(N*oh*ow, -1), oh, ow

# ── Conv2D — forward ──────────────────────────────
class Conv2D:
    def forward(self, x):
        col, oh, ow = im2col(x, k, k, stride, pad)  # 展開
        W_col = self.W.reshape(out_ch, -1)            # (out_ch, C·k·k)
        out = col @ W_col.T + self.b                  # 行列積 → 畳み込み完成
        return out.reshape(N, oh, ow, -1).transpose(0,3,1,2)

# ── ResNet Skip Connection ─────────────────────────
class ResBlock:
    def forward(self, x):
        h = self.conv2.forward(self.relu1.forward(self.conv1.forward(x)))
        return self.relu2.forward(h + x)   # ← Skip: F(x) + x
    def backward(self, grad):
        grad = self.relu2.backward(grad)
        skip = grad.copy()                # 分岐した勾配
        grad = self.conv1.backward(self.relu1.backward(self.conv2.backward(grad)))
        return grad + skip                 # 恒等写像を通した勾配を加算

# ── LSTM Cell: 4ゲートをまとめて1つの行列で処理 ──
class LSTMCell:
    def forward(self, x, h_prev, c_prev):
        xh = np.hstack([x, h_prev])           # 入力 + 前の隠れ状態
        gates = xh @ self.W + self.b             # 全ゲートを一括計算

        f = sigmoid(gates[:, :H])               # 忘却ゲート: 過去を何%残す
        i = sigmoid(gates[:, H:2*H])            # 入力ゲート: 何%取り込む
        g = np.tanh(gates[:, 2*H:3*H])          # セル候補: 新しい情報
        o = sigmoid(gates[:, 3*H:])             # 出力ゲート: 何%出力する

        c = f * c_prev + i * g   # セル状態 (長期記憶) C_t = f⊙C_{t-1} + i⊙g
        h = o * np.tanh(c)      # 隠れ状態 (短期記憶) h_t = o⊙tanh(C_t)
        return h, c

# ── Scaled Dot-Product Attention ─────────────────
def scaled_dot_product_attention(Q, K, V, mask=None):
    d_k    = Q.shape[-1]
    scores = Q @ K.swapaxes(-2, -1) / np.sqrt(d_k)  # QK^T / √d_k
    if mask is not None:
        scores = np.where(mask, scores, -1e9)    # 未来トークンをマスク
    attn = softmax_2d(scores)                     # Attention 重み
    return attn @ V, attn                          # 加重和

class RoPE:
    def __init__(self, d_k, max_len=512):
        theta = 1.0 / (10000 ** (np.arange(0, d_k, 2) / d_k))

month6_generative_rl.py — VAE Reparameterization Trick
# ── VAE: Reparameterization Trick ────────────────
class VAE:
    def encode(self, x):
        h  = relu(x @ self.We1 + self.be1)
        mu = h @ self.Wmu + self.bmu         # 潜在分布の平均 μ
        lv = h @ self.Wlv + self.blv         # 潜在分布の log σ²
        return mu, lv, h

    def reparameterize(self, mu, log_var):
        eps = np.random.randn(*mu.shape)    # ε ~ N(0,I)
        std = np.exp(0.5 * log_var)
        return mu + std * eps               # z = μ + σ·ε (微分可能!)

    def loss(self, x, x_hat, mu, log_var):
        recon = -np.mean(x * np.log(x_hat+1e-8) + (1-x)*np.log(1-x_hat+1e-8))
        # KL(q(z|x) || p(z)) = -0.5 Σ(1 + log σ² - μ² - σ²)
        kl = -0.5 * np.mean(1 + log_var - mu**2 - np.exp(log_var))
        return recon + kl, recon, kl  # 再構成誤差 + KL正則化
